# Три этапа game5.sh (спойлеры). Путь к игре вместо GAME (например terminal_game_v1.5).

# --- Уровень 1 — каталог: GAME/level_1 ---
comm -12 <(grep -oE 'cd sector_[0-9]+' shell_history.log | sed 's/^cd //' | sort -u) <(sort available_sectors.db)

# --- Уровень 2 — каталог: GAME/level_1/<ответ_уровня_1>/ (рядом позже появятся vpn_*.log; сейчас только shadow_fragment.db) ---
HDR=$(head -1 shadow_fragment.db) && SEP="${HDR#type}" && SEP="${SEP%%date*}" && awk -F"$SEP" 'NR>1 && $15=="operator" {print $11; exit}' shadow_fragment.db

# --- Уровень 3 — тот же каталог (уже есть vpn_*.log) ---
awk 'match($0, /sess=sess[0-9]+/) { id = substr($0, RSTART, RLENGTH); c[id]++ } END { for (i in c) if (c[i] == 2) print i }' vpn_*.log
SID=$(awk 'match($0, /sess=sess[0-9]+/) { id = substr($0, RSTART, RLENGTH); c[id]++ } END { for (i in c) if (c[i] == 2) print i }' vpn_*.log | head -1)
grep -hF "$SID" vpn_*.log | grep 'trace_exit_node=' | sed 's/.*trace_exit_node=\([a-z0-9]*\).*/\1/'
